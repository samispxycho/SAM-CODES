//Some methods are common in all the Data Structures..those are

interface Collection{
    boolean add();
    boolean addAll();

    boolean remove();
    boolean removeAll();

    void clear();

    boolean contains();
    boolean containsAll();

    boolean isEmpty();
    
    int size();     ---etc
}

interface List extends Collection{
    void add();
    void addAll();

    Object get(int index);

    Object remove(int index);

    set(int index, Object value);

    int indexof(value);
    int lastIndexOf(value); ----etc;
}
